CREATE TABLE users (id bigint NOT NULL PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
                    name varchar(30) NOT NULL,
                    email varchar(254) NOT NULL,
                    email_verified_at timestamp with time zone NOT NULL,
                    password varchar(70) NOT NULL,
                    phone_number varchar(12) NOT NULL);


-- Create model Rooms

--

CREATE TABLE rooms (id bigint NOT NULL PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
                    home_type varchar(30) NOT NULL,
                    address varchar(50) NOT NULL,
                    has_tv boolean NOT NULL,
                    has_internet boolean NOT NULL,
                    has_kitchen boolean NOT NULL,
                    has_air_con boolean NOT NULL,
                    price integer NOT NULL,
                    latitude double precision NOT NULL,
                    longnitude double precision NOT NULL);

--

-- Create model Reservations

--

CREATE TABLE reservations (id bigint NOT NULL PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
                           start_date timestamp with time zone NOT NULL,
                           end_date timestamp with time zone NOT NULL,
                           price integer NOT NULL,
                           total integer NOT NULL);
--

-- Create model Reviews

CREATE TABLE reviews(id bigint NOT NULL PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
                     rating integer NOT NULL,
                     reservation_id bigint NOT NULL);
